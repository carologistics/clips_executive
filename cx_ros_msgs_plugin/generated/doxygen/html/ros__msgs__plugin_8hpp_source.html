56 struct MessageInfo
57 {
58 std::shared_ptr<void> external_created_ptr;
59 void * msg_ptr;
60 const rosidl_typesupport_introspection_cpp::MessageMembers * members;
61 bool is_sub_msg;
62 rcutils_allocator_t allocator;
63
64 MessageInfo(
65 const rosidl_typesupport_introspection_cpp::MessageMembers * members,
66 void * parent_ptr = nullptr)
67 : msg_ptr(nullptr), members(members)
68 {
69 is_sub_msg = (parent_ptr != nullptr);
70 if (parent_ptr) {
71 msg_ptr = parent_ptr;
72 } else {
73 // Allocate memory for the message
74 allocator = rcutils_get_default_allocator();
75 msg_ptr = allocator.zero_allocate(1, members->size_of_, allocator.state);
76 if (members->init_function) {
77 members->init_function(
78 msg_ptr,
79 rosidl_runtime_cpp::MessageInitialization::ALL); // Full
80 // initialization
81 }
82 }
83 }
84 MessageInfo(
85 const rosidl_typesupport_introspection_cpp::MessageMembers * members,
86 std::shared_ptr<void> external_created_ptr)
87 : external_created_ptr(external_created_ptr),
88 msg_ptr(external_created_ptr.get()),
89 members(members)
90 {
91 is_sub_msg = true;
92 }
93
94 ~MessageInfo()
95 {
96 if (!is_sub_msg) {
97 // Finalize the message and free memory
98 if (members && members->fini_function) {
99 members->fini_function(msg_ptr);
100 }
101 allocator.deallocate(msg_ptr, allocator.state);
102 }
103 }
104
105 // Disable copy constructor and copy assignment to prevent double freeing
106 MessageInfo(const MessageInfo &) = delete;
107 MessageInfo & operator=(const MessageInfo &) = delete;
108
109 // Enable move constructor and move assignment for efficiency
110 MessageInfo(MessageInfo && other) noexcept
111 : external_created_ptr(other.external_created_ptr),
112 msg_ptr(other.msg_ptr),
113 members(other.members),
114 is_sub_msg(other.is_sub_msg),
115 allocator(other.allocator)
116 {
117 RCLCPP_WARN(rclcpp::get_logger("MESSAGE INFO"), "ASSIGN");
118 other.msg_ptr = nullptr;
119 other.external_created_ptr.reset();
120 other.members = nullptr;
121 other.is_sub_msg = false;
122 }
123 };
124 struct MessageInfoHasher
125 {
126 std::size_t operator()(const MessageInfo & key) const
127 {
128 return std::hash<void *>()(key.external_created_ptr.get()) ^
129 std::hash<void *>()(key.msg_ptr) ^
130 std::hash<const rosidl_typesupport_introspection_cpp::MessageMembers *>()(key.members);
131 }
132 };
